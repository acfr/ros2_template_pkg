40 /* This example creates a subclass of Node and uses std::bind() to register a
41 * member function as a callback from the timer. */
42 
43 class ElapsedTimePublisher : public rclcpp::Node
44 {
45  public:
46  ElapsedTimePublisher()
47  : Node("time_publisher"), count_(1)
48  {
49  publisher_ = this->create_publisher<std_msgs::msg::String>("time", 10);
50  timer_ = this->create_wall_timer(
51  1000ms, std::bind(&ElapsedTimePublisher::timer_callback, this));
52  }
53 
54  private:
55  void timer_callback()
56  {
57  auto message = std_msgs::msg::String();
58  message.data = std::to_string(count_++);
59  RCLCPP_INFO(this->get_logger(), "'%s' seconds", message.data.c_str());
60  publisher_->publish(message);
61  }
62  rclcpp::TimerBase::SharedPtr timer_;
63  rclcpp::Publisher<std_msgs::msg::String>::SharedPtr publisher_;
64  size_t count_;
65 };
