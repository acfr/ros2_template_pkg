67  void send_goal()
68  {
69  using namespace std::placeholders;
70 
71  this->timer_->cancel();
72 
73  this->goal_done_ = false;
74 
75  if (!this->client_ptr_) {
76  RCLCPP_ERROR(this->get_logger(), "Action client not initialized");
77  }
78 
79  if (!this->client_ptr_->wait_for_action_server(std::chrono::seconds(10))) {
80  RCLCPP_ERROR(this->get_logger(), "Action server not available after waiting");
81  this->goal_done_ = true;
82  return;
83  }
84 
85  auto goal_msg = Fibonacci::Goal();
86  goal_msg.order = 10;
87 
88  RCLCPP_INFO(this->get_logger(), "Sending goal");
89 
90  auto send_goal_options = rclcpp_action::Client<Fibonacci>::SendGoalOptions();
91  send_goal_options.goal_response_callback =
92  std::bind(&MinimalActionClient::goal_response_callback, this, _1);
93  send_goal_options.feedback_callback =
94  std::bind(&MinimalActionClient::feedback_callback, this, _1, _2);
95  send_goal_options.result_callback =
96  std::bind(&MinimalActionClient::result_callback, this, _1);
97  auto goal_handle_future = this->client_ptr_->async_send_goal(goal_msg, send_goal_options);
98  }
99

103  bool goal_done_;

105  void goal_response_callback(GoalHandleFibonacci::SharedPtr goal_handle)
106  {
107  if (!goal_handle) {
108  RCLCPP_ERROR(this->get_logger(), "Goal was rejected by server");
109  } else {
110  RCLCPP_INFO(this->get_logger(), "Goal accepted by server, waiting for result");
111  }
112  }
113 
114  void feedback_callback(
115  GoalHandleFibonacci::SharedPtr,
116  const std::shared_ptr<const Fibonacci::Feedback> feedback)
117  {
118  RCLCPP_INFO(
119  this->get_logger(),
120  "Next number in sequence received: %" PRId32,
121  feedback->sequence.back());
122  }
123 
124  void result_callback(const GoalHandleFibonacci::WrappedResult & result)
125  {
126  this->goal_done_ = true;
127  switch (result.code) {
128  case rclcpp_action::ResultCode::SUCCEEDED:
129  break;
130  case rclcpp_action::ResultCode::ABORTED:
131  RCLCPP_ERROR(this->get_logger(), "Goal was aborted");
132  return;
133  case rclcpp_action::ResultCode::CANCELED:
134  RCLCPP_ERROR(this->get_logger(), "Goal was canceled");
135  return;
136  default:
137  RCLCPP_ERROR(this->get_logger(), "Unknown result code");
138  return;
139  }
140 
141  RCLCPP_INFO(this->get_logger(), "Result received");
142  for (auto number : result.result->sequence) {
143  RCLCPP_INFO(this->get_logger(), "%" PRId32, number);
144  }
145  }
