63  rclcpp_action::GoalResponse handle_goal(
64  const rclcpp_action::GoalUUID & uuid,
65  std::shared_ptr<const Fibonacci::Goal> goal)
66  {
67  RCLCPP_INFO(this->get_logger(), "Received goal request with order %d", goal->order);
68  (void)uuid;
69  // Let's reject sequences that are over 9000
70  if (goal->order > 9000) {
71  return rclcpp_action::GoalResponse::REJECT;
72  }
73  return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
74  }
75 
76  rclcpp_action::CancelResponse handle_cancel(
77  const std::shared_ptr<GoalHandleFibonacci> goal_handle)
78  {
79  RCLCPP_INFO(this->get_logger(), "Received request to cancel goal");
80  (void)goal_handle;
81  return rclcpp_action::CancelResponse::ACCEPT;
82  }
83 
84  void execute(const std::shared_ptr<GoalHandleFibonacci> goal_handle)
85  {
86  RCLCPP_INFO(this->get_logger(), "Executing goal");
87  rclcpp::Rate loop_rate(1);
88  const auto goal = goal_handle->get_goal();
89  auto feedback = std::make_shared<Fibonacci::Feedback>();
90  auto & sequence = feedback->sequence;
91  sequence.push_back(0);
92  sequence.push_back(1);
93  auto result = std::make_shared<Fibonacci::Result>();
94 
95  for (int i = 1; (i < goal->order) && rclcpp::ok(); ++i) {
96  // Check if there is a cancel request
97  if (goal_handle->is_canceling()) {
98  result->sequence = sequence;
99  goal_handle->canceled(result);
100  RCLCPP_INFO(this->get_logger(), "Goal Canceled");
101  return;
102  }
103  // Update sequence
104  sequence.push_back(sequence[i] + sequence[i - 1]);
105  // Publish feedback
106  goal_handle->publish_feedback(feedback);
107  RCLCPP_INFO(this->get_logger(), "Publish Feedback");
108 
109  loop_rate.sleep();
110  }
111 
112  // Check if goal is done
113  if (rclcpp::ok()) {
114  result->sequence = sequence;
115  goal_handle->succeed(result);
116  RCLCPP_INFO(this->get_logger(), "Goal Succeeded");
117  }
118  }
119 
120  void handle_accepted(const std::shared_ptr<GoalHandleFibonacci> goal_handle)
121  {
122  using namespace std::placeholders;
123  // this needs to return quickly to avoid blocking the executor, so spin up a new thread
124  std::thread{std::bind(&MinimalActionServer::execute, this, _1), goal_handle}.detach();
125  }
126 }; // class MinimalActionServer
