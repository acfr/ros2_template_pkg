38 def generate_launch_description():

43  turtlesim_ns_launch_arg = DeclareLaunchArgument(
44  'turtlesim_ns',

46  )

48  'use_provided_red',
49  default_value='False'

52  'new_background_r',
53  default_value='200'
54  )
55 
56  turtlesim_node = Node(
57  package='turtlesim',
58  namespace=turtlesim_ns,
59  executable='turtlesim_node',
60  name='sim'
61  )
62  spawn_turtle = ExecuteProcess(
63  cmd=[[
64  'ros2 service call ',
65  turtlesim_ns,
66  '/spawn ',
67  'turtlesim/srv/Spawn ',
68  '"{x: 2, y: 2, theta: 0.2}"'
69  ]],
70  shell=True
71  )
72  change_background_r = ExecuteProcess(
73  cmd=[[
74  'ros2 param set ',
75  turtlesim_ns,
76  '/sim background_r ',
77  '120'
78  ]],
79  shell=True
80  )
81  change_background_r_conditioned = ExecuteProcess(

87  use_provided_red
88  ])
89  ),
90  cmd=[[
91  'ros2 param set ',
92  turtlesim_ns,
93  '/sim background_r ',
94  new_background_r
95  ]],
96  shell=True
97  )
98 
99  return LaunchDescription([

101  use_provided_red_launch_arg,
102  new_background_r_launch_arg,
103  turtlesim_node,
104  spawn_turtle,
105  change_background_r,
